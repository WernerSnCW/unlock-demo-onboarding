How to compute (drop-in steps inside simulateV2)

Assume you already compute:

blended 12-month mean shocks per bucket (mean12[b])

horizon-scaled per-bucket returns RH[b] = (1+mean12[b])^(H/12) − 1

contributions (dot product) and fan (p05/50/95) from MC or deterministic means

Add the following after you build fan:

// 1) £ impact & range (always)
const last = fan[fan.length - 1];
const p50C = last.current.p50;
const p50T = last.target.p50;
res.endValue = { current: p50C, target: p50T, diffGBP: p50T - p50C };
res.endValueBand = { current: last.current, target: last.target };

// 2) Breakeven month (median)
res.breakevenMonthMed = null;
for (const pt of fan) { if (pt.target.p50 >= pt.current.p50) { res.breakevenMonthMed = pt.t; break; } }

// 3) Downside risk (only if MC ON and you have final return arrays)
if (mcFinalReturnsCurrent && mcFinalReturnsTarget) {
  const prob = (arr:number[]) => arr.filter(x => x < 0).length / arr.length;
  const es5 = (arr:number[]) => {
    const s = [...arr].sort((a,b)=>a-b);
    const k = Math.max(1, Math.floor(s.length * 0.05));
    const worst = s.slice(0, k);
    return worst.reduce((a,x)=>a+x,0) / worst.length;
  };
  res.downside = {
    probLoss: { current: prob(mcFinalReturnsCurrent), target: prob(mcFinalReturnsTarget) },
    es5:      { current: es5(mcFinalReturnsCurrent), target: es5(mcFinalReturnsTarget) }
  };
}

// 4) After-costs uplift (optional passthrough from request.costs)
if (req.costs) {
  res.costs = {
    estTurnoverPp: req.costs.estTurnoverPp,
    estCostPct:    req.costs.estCostPct,
    diffAfterCostsPp: (res.expectedReturnTarget - res.expectedReturnCurrent) - req.costs.estCostPct
  };
}

// 5) Difference attribution (factor buckets)
const FACTORS: Record<string, string[]> = {
  "Equities": ["GLOBAL_EQUITY","UK_EQUITY_VALUE","GROWTH_TECH"],
  "Credit/Income": ["IG_CREDIT"],
  "Duration": ["GILTS_LONG"],
  "Liquidity": ["CASH","BILLS_SHORT_GILTS"],
  "Real Assets": ["GOLD","COMMODITIES","PROPERTY_UK_RESI"],
  "Other": ["ALTERNATIVES","COLLECTIBLES_ART","COLLECTIBLES_WINE","CRYPTO_BTC","CRYPTO_ETH"]
};
res.diffAttribution = Object.entries(FACTORS).map(([name, bs])=>{
  const pp = bs.reduce((s,b)=> s + ((tgt[b]-cur[b]) * (RH[b] || 0)), 0);
  return { factor: name, pp };
}).sort((a,b)=> Math.abs(b.pp) - Math.abs(a.pp));

// 6) Stresses (pure scenarios, headline only)
res.stresses = ["S003","S007","S009"].map(id => {
  // Reuse same engine with scenarioWeights = { [id]: 1 }
  const RHstress: Record<string, number> = computeRHForPureScenario(id); // same RH formula with mean12 from that scenario
  const retC = CANONICAL_BUCKETS.reduce((s,b)=> s + cur[b]*(RHstress[b]||0), 0);
  const retT = CANONICAL_BUCKETS.reduce((s,b)=> s + tgt[b]*(RHstress[b]||0), 0);
  return { id, label: SCENARIO_LABELS[id] || id, retCurrent: retC, retTarget: retT, diffPp: retT - retC };
});
