// beliefIndependent.ts
export type Scenario = string;
export type QuestionID = string;

type Weights = Record<QuestionID, Record<Scenario, number>>;

export interface RegimeCfg {
  weightMultiplier: number;   // e.g. low 0.8, normal 1.0, high 1.4, crisis 1.8
  months: number;             // horizon for decay
}

export type DecayRates = Record<Scenario, number>; // annualized λ, e.g. { "Energy Shock": 0.25, ... }
export type CorrMap = Record<string, number>;      // "QID|QID" -> correlation in [-1,1]

/** Likert (1..5) -> centered z in [-1,+1], 3 => 0 */
function likertZ(answer: number): number {
  return Math.max(-1, Math.min(1, (answer - 3) / 2));
}

/** direction: "higher->..." or "lower->..." -> positive risk signal in [0,1] */
function riskSignal(answer: number, direction: string): number {
  const [side] = direction.split("->");
  const z = likertZ(answer);
  const s = side.trim().toLowerCase() === "lower" ? -z : z; // flip if "lower"
  return Math.max(0, s); // only supportive signal
}

/** simple correlation dampening: si' = si / (1 + 0.5 * Σ|corr(i,j)| for j with s_j>0) */
function dampSignals(
  rawSignals: Record<QuestionID, number>,
  corr: CorrMap
): Record<QuestionID, number> {
  const out: Record<QuestionID, number> = {};
  for (const qi of Object.keys(rawSignals)) {
    const si = rawSignals[qi];
    if (si <= 0) { out[qi] = 0; continue; }
    let tot = 0;
    for (const key of Object.keys(corr)) {
      const [a, b] = key.split("|");
      if (a === qi && rawSignals[b] > 0) tot += Math.abs(corr[key]);
      if (b === qi && rawSignals[a] > 0) tot += Math.abs(corr[key]);
    }
    const factor = 1 / (1 + 0.5 * tot);
    out[qi] = si * factor;
  }
  return out;
}

/** Aggregate adjusted signals into per-scenario raw scores */
function aggregateScores(
  adjSignals: Record<QuestionID, number>,
  weights: Weights
): Record<Scenario, number> {
  const scores: Record<Scenario, number> = {};
  for (const qid of Object.keys(adjSignals)) {
    const s = adjSignals[qid];
    if (s <= 0) continue;
    const wm = weights[qid] || {};
    for (const scen of Object.keys(wm)) {
      scores[scen] = (scores[scen] || 0) + s * wm[scen]; // weight can be ±
    }
  }
  return scores;
}

/** Apply regime multiplier and exponential time decay */
function applyRegimeAndDecay(
  scores: Record<Scenario, number>,
  regime: RegimeCfg,
  decay: DecayRates
): Record<Scenario, number> {
  const out: Record<Scenario, number> = {};
  for (const scen of Object.keys(scores)) {
    const λ = decay[scen] ?? 0;
    const decayFactor = Math.exp(-λ * (regime.months / 12));
    out[scen] = scores[scen] * regime.weightMultiplier * decayFactor;
  }
  return out;
}

/**
 * Convert scores to independent probabilities in [0,1].
 * Use Poisson-complement: p = 1 - exp(-k * max(0,score)).
 * k is a sensitivity knob (start with 1.0).
 */
export function independentProbs(
  scores: Record<Scenario, number>,
  k = 1.0
): Record<Scenario, number> {
  const out: Record<Scenario, number> = {};
  for (const scen of Object.keys(scores)) {
    const x = Math.max(0, scores[scen]);
    out[scen] = 1 - Math.exp(-k * x);
  }
  return out; // NOT normalised; multiple can be high
}

/** Main: answers -> independent scenario probabilities */
export function computeIndependentProbabilities(opts: {
  answers: Record<QuestionID, number>;          // 1..5
  qMeta: Record<QuestionID, { direction: string }>;
  weights: Weights;
  regime: RegimeCfg;
  decay: DecayRates;
  corr: CorrMap;
  k?: number;                                   // optional sensitivity (default 1.0)
}): Record<Scenario, number> {
  const { answers, qMeta, weights, regime, decay, corr, k = 1.0 } = opts;

  // raw signals
  const raw: Record<QuestionID, number> = {};
  for (const qid of Object.keys(answers)) {
    const meta = qMeta[qid];
    if (!meta) continue;
    raw[qid] = riskSignal(answers[qid], meta.direction);
  }

  const adjusted = dampSignals(raw, corr);
  const scores = aggregateScores(adjusted, weights);
  const s2 = applyRegimeAndDecay(scores, regime, decay);
  return independentProbs(s2, k); // <-- independent [0,1], no softmax
}

// scenarioBlender.ts (patch)

export function computePairWeights(
  probs: Record<ScenarioName, number>,
  pairCap = 0.8 // allow larger cap since probs are independent now
) {
  const names = Object.keys(probs);
  const raw: Record<string, number> = {};
  let W = 0;

  for (let i = 0; i < names.length; i++) {
    for (let j = i + 1; j < names.length; j++) {
      const si = names[i], sj = names[j];
      const wij = (probs[si] ?? 0) * (probs[sj] ?? 0); // <-- independent probs
      if (wij > 0) {
        raw[pairKey(si, sj)] = wij;
        W += wij;
      }
    }
  }
  const W_pair = Math.min(W, pairCap);
  const normalised: Record<string, number> = {};
  if (W > 0 && W_pair > 0) {
    for (const k of Object.keys(raw)) {
      normalised[k] = (raw[k] / W) * W_pair;
    }
  }
  return { pairWeights: normalised, W_pair };
}

def compute_independent_probabilities(master: dict,
                                      answers: dict,
                                      regime: str = "normal",
                                      months: int = 12,
                                      corr_matrix: dict = None,
                                      decay: dict = None,
                                      k: float = 1.0) -> dict:
    """
    Independent scenario probabilities in [0,1], no sum=1 constraint.
    p = 1 - exp(-k * score_after_regime_decay)
    """
    if decay is None:
        from enhanced_belief_engine import DEFAULT_DECAY as DD
        decay = DD.copy()
    from enhanced_belief_engine import DEFAULT_REGIMES
    rcfg = DEFAULT_REGIMES[regime]

    # build raw signals
    qmeta = {q["id"]: q for q in master["questions"]}
    raw = {}
    for qid, ans in answers.items():
        if qid not in qmeta:
            continue
        dirn = qmeta[qid]["direction"]
        z = (ans - 3.0)/2.0
        s = -z if dirn.split("->",1)[0].strip().lower()=="lower" else z
        raw[qid] = max(0.0, s)

    # correlation
    def corr_dampen(signals, corr):
        out = {}
        for qi, si in signals.items():
            if si <= 0: out[qi]=0; continue
            tot = 0
            for (a,b), c in (corr or {}).items():
                if qi==a and signals.get(b,0)>0: tot += abs(c)
                if qi==b and signals.get(a,0)>0: tot += abs(c)
            out[qi] = si * (1.0/(1.0+0.5*tot))
        return out

    adj = corr_dampen(raw, corr_matrix)

    # aggregate
    weights = {q["id"]: q["weights"] for q in master["questions"]}
    scores = {}
    for qid, s in adj.items():
        for scen, w in weights.get(qid, {}).items():
            scores[scen] = scores.get(scen,0.0) + s*w

    # regime + decay
    s2 = {}
    for scen, val in scores.items():
        lam = decay.get(scen, 0.0)
        decay_factor = math.exp(-lam * (months/12))
        s2[scen] = val * rcfg.weight_multiplier * decay_factor

    # independent probs
    probs = {scen: 1 - math.exp(-k * max(0.0, val)) for scen, val in s2.items()}
    return probs

3) What changes in behavior (and why it’s better for overlaps)
Before (softmax): Σp=1 forced scenarios to compete. If “Debt Spiral” went up, “Energy Shock” had to go down — unrealistic when both can co-exist.


Now (independent): Each scenario has its own likelihood 0,1. Several can be high at once.


The blender turns those co-existing probabilities into correlated, not compounded asset impacts using:


pair weights p_i * p_j (how much those two co-occur),


overlap tables (or synthetic overlaps),


and a cap on total pair mass (so overlaps can’t dominate).
